import java.util.Scanner;

public class TemperaturRechner {
    public static void main(String[] args) {
        // Scanner-Objekt erstellen
        Scanner scanner = new Scanner(System.in);

        try {
            // Eingabeaufforderung anzeigen
            System.out.print("Gib die Temperatur ein: ");

            // Eingabe von der Konsole lesen
            String eingabeString = scanner.nextLine();

            // Eingabe in eine Zahl umwandeln
            int temperatur = Integer.parseInt(eingabeString);

            // 5 zur Temperatur hinzufügen
            int ergebnis = temperatur + 5;

            // Ergebnis anzeigen
            System.out.println("Die neue Temperatur nach Addition von 5 ist: " + ergebnis);
        } catch (NumberFormatException e) {
            // Fehlermeldung anzeigen, wenn die Eingabe keine gültige Zahl war
            System.out.println("Bitte gib eine gültige Zahl ein.");
        }

        // Scanner schließen
        scanner.close();
    }
}
