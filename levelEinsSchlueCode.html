Der Code für den Schlüsselschrank ist durcheinander geraten. Kannst Du das Programm wieder zum laufen bringen?


public class SchluesselSchrank {
    public static void main(String[] args) {
        // Variablen deklarieren und initialisieren
        int x = 10;
        double y = 3.5;
        int z = 2;
        int sum = x + y; 
        double division = x / z;  

        int modulo = y % z; 
        int result = sum * 100 + (int)division * 10 + modulo;  
        // Ergebnis ausgeben
        System.out.println("Der Code für den Schlüsselschrank ist: " + result);
    }
}
