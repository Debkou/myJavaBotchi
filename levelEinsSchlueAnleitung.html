Vervollständige den Code um die Ausgabe zu entschlüsseln


import java.util.Scanner;

public class NachrichtEntschluesseln {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        // Erwarteter Code
        String correctCode = " ";

        // Eingabe des Codes vom Benutzer
        System.out.print("Bitte geben Sie den Code ein: ");
        String inputCode = scanner.nextLine();

        // Überprüfen des Codes
        if (correctCode.equals(inputCode)) {
            // Wenn der Code korrekt ist, führe die Methode aus
            String text = "Zlu vwxglhuhq dq ghu IKVZI";
            int shift = -3; // Anzahl der Positionen nach links verschieben

            String encryptedText = encrypt(text, shift);
            System.out.println("Original Text: " + text);
            System.out.println("Verschlüsselter Text: " + encryptedText);
        } else {
            // Wenn der Code nicht korrekt ist, gebe eine Fehlermeldung aus
            System.out.println("Code nicht korrekt");
        }

        // Scanner schließen
        scanner.close();
    }

    public static String encrypt(String text, int shift) {
        StringBuilder result = new StringBuilder();
        shift = shift % 26; // Um sicherzustellen, dass der Shift innerhalb des Alphabets bleibt

        for (char i : text.toCharArray()) {
            if (Character.isLetter(i)) {
                char base = Character.isLowerCase(i) ? 'a' : 'A';
                char shiftedChar = (char) ((i - base + shift + 26) % 26 + base);
                result.append(shiftedChar);
            } else {
                // Wenn es kein Buchstabe ist, bleibt das Zeichen unverändert
                result.append(i);
            }
        }

        return result.toString();
    }
}
