public class NachrichtEntschluesseln {
    public static void main(String[] args) {
      
        int code 

        String verschluesselt = "Xms vwyfpenfto ep fskz GIPYG";

        // Umwandlung des Codes in eine Zeichenkette
        String codeStr = String.valueOf(code);

        // Initialisierung der Variablen für das Entschlüsseln
        String entschluesselt = "";
        int shift = 0;

        // Schleife über den verschlüsselten Text
        for (int i = 0; i < verschluesselt.length(); i++) {
            char c = verschluesselt.charAt(i);

            // Berechnung des Shifts
            shift = Character.getNumericValue(codeStr.charAt(i % codeStr.length()));

            // Entschlüsseln des Zeichens
            if (Character.isLetter(c)) {
                if (Character.isUpperCase(c)) {
                    c = (char)('A' + (c - 'A' - shift + 26) % 26);
                } else {
                    c = (char)('a' + (c - 'a' - shift + 26) % 26);
                }
            }

            entschluesselt += c;
        }

        // Ergebnis ausgeben
        System.out.println("Die entschlüsselte Nachricht ist: " + entschluesselt);
    }
}
